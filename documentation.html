5. المسارات (Routes)

// POS
GET  /pos                    → شاشة POS
GET  /pos/products            → جلب المنتجات (AJAX)
GET  /pos/products/search     → البحث (AJAX)
GET  /pos/products/barcode/{id} → بالباركود (AJAX)

// Sales
POST /sales                  → حفظ بيع
POST /sales/suspend          → تعليق بيع
GET  /suspended-sales        → المبيعات المعلقة
POST /suspended-sales/{id}/resume → استئناف

// Resources
Route::resource('products');
Route::resource('categories');
Route::resource('customers');
Route::resource('suppliers');
Route::resource('expenses');

6. شاشة POS التقنية

6.1 الهيكل

graph TB A[POS] --> B[Header: البحث + العميل] A --> C[Left: المنتجات] A --> D[Right: السلة] C --> C1[تصنيفات] C --> C2[شبكة المنتجات] D --> D1[عناصر السلة] D --> D2[المجموع+الضريبة] D --> D3[أزرار الدفع]

6.2_functions JavaScript

loadProducts(categoryId)    // تحميل المنتجات
addToCartDirect(id,name,price,stock)  // إضافة للسلة
renderCart()                // رسم السلة
updateTotals()              // تحديث المجموع
openPaymentModal()          // فتح نافذة الدفع
processPayment()            // معالجة الدفع
suspendSale()               // تعليق البيع
clearCart()                 // مسح السلة

6.3 Keyboard Shortcuts

F2 - دفع | F4 - تعليق | Esc - مسح
  • الضريبة: 15% (تعديل في TAX_RATE)
  • الدفع: نقدي - بطاقة - تقسيم
  • البحث: بالاسم أو الباركود
